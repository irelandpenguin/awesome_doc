     7	func itoa(val int) string { // do it here rather than with fmt to avoid dependency
     8		if val < 0 {
     9			return "-" + itoa(-val)
    10		}
    11		var buf [32]byte // big enough for int64
    12		i := len(buf) - 1
    13		for val >= 10 {
    14			buf[i] = byte(val%10 + '0')
    15			i--
    16			val /= 10
    17		}
    18		buf[i] = byte(val + '0')
    19		return string(buf[i:])
    20	}
