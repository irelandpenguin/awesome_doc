     7	// Copysign(x, y) returns a value with the magnitude
     8	// of x and the sign of y.
     9	func Copysign(x, y float64) float64 {
    10		const sign = 1 << 63
    11		return Float64frombits(Float64bits(x)&^sign | Float64bits(y)&sign)
    12	}
